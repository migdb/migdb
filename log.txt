+	return gen.isColumnInTable(self.owningSchemaName, self.owningTableName, self.name)

+	// pro dan sloupec neexistuje tabulkov omezen 'ForeignKey'

+	var columns : OrderedSet(RDB::Column);

+	return false;

+	self.findTable(sName, tName).constraints.oclAsType(RDB::PrimaryKey)->forEach(r) {

+	return false;

+query RDB::ModelGeneration::isForeignKey(sName : String, tName : String, cName : String) : Boolean {
+	// vybere omezen typu 'ForeignKey'
+	self.findTable(sName, tName).constraints.oclAsType(RDB::ForeignKey)->forEach(r) {
+		return r.constrainedColumn.name = cName;
+	};
+	return false;

+query isValidIdentifier(identifier : String):Boolean{
+	var valid : Boolean := true;
+	if(not beginsWithChar(identifier))then{
+			log("Model invalid - identifier " + identifier + " not begins with char");
+			valid := false;
+		}endif;
+		if(isKeyWord(identifier))then{
+			log("Model invalid - identifier " + identifier + " is Postgresql keyWord");
+			valid := false;
+		}endif;
+		if(not containsAllowedChars(identifier))then{
+			log("Model invalid - identifier " + identifier + " contains unallowed characters");		
+			valid := false;
+		}endif;
+		return valid;
+}

+query beginsWithChar(identifier : String ):Boolean{
+	var firstChar: String  := identifier.substring(0,1).toLower() ;
+	return isLatinCharacter(firstChar);
+}

+		char = "f" or char = "g" or char = "h" or char = "i" or char = "j" or
+		 char = "k" or char = "l" or char = "m" or char = "n" or char = "o" or
+		 char = "p" or char = "q" or char = "r" or char = "s" or char = "t" or
+		  char = "u" or char = "v" or char = "w" or char = "x" or char = "y" or
+		   char = "z")then{
+		return true;
+	}endif;
+	return false;
+}

+query isDigit(char : String) : Boolean{
+	if(char = "0" or char = "1" or char = "2" or char = "3" or char = "4" or 
+		char = "5" or char = "6" or char = "7" or char = "8" or char = "9")then{
+		return true;
+	}endif;
+	return false;

+query containsAllowedChars(identifier : String):Boolean{
+	var position:Integer := 0;
+	var lowerIdentifier := identifier.toLower();
+	while(position < lowerIdentifier.size()){
+		var char:String := lowerIdentifier.substring(position, position+1);
+		if(not (isLatinCharacter(char) or char = "_" or isDigit(char)))then {
+			return false;
+		}endif;
+		position = position + 1;
+	};
+	return true;
+}
